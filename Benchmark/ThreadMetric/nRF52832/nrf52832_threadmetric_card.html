Application code — any board, no DTS, no Kconfig

// nRF52832 · static allocation · always-on stack guard + null check
static uint8_t gStack[TAKTOS_THREAD_MEM_SIZE(512)];
static TaktOSSem_t gSem;

void sensorTask(void *) {
    for (;;) {
        TaktOSSemTake(&gSem, true, TAKTOS_WAIT_FOREVER);
        process();
    }
}

int main() {
    TaktOSInit(64'000'000, 1000, TAKTOS_TICK_CLOCK_PROCESSOR, 0);
    TaktOSSemInit(&gSem, 0, 1);
    TaktOSThreadCreate(gStack, sizeof(gStack), sensorTask, nullptr, 16);
    TaktOSStart(); // never returns
}
